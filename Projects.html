# This program calculates the money collected from a fundraiser
# and writes it to a file
# Landon Harl
# 10/13/22

def main():
    cTotal = 0
    playAgain = "yes"
    fundraiser = init()
    while (playAgain == "yes"):
        iName = getName()
        iId = studentID()
        iFunds = FRAmt()
        iMajor = Major()
        cTotal = calcs(iFunds, cTotal)
        playAgain = input("Would you like to enter more? (yes/no) ")
    output(iName, iId, iFunds, iMajor, fundraiser, cTotal)
    closing(fundraiser)

def calcs(iFunds, cTotal):
    cTotal = cTotal + iFunds
    return cTotal

def init():
    try:
        fundraiser = open(r'c:\Python\fundraiser.dat', 'w')
    except:
        print("File Error. Program Terminated.")
        exit(0)
    return fundraiser


def getName():
    valid = False
    while valid == False:
        iName = (input("Enter name: "))
        if iName.strip() != "" and iName.isalpha():
            valid = True
        else:
            print("Error. Name is required")
    return iName

def studentID():
    valid = False
    while valid == False:
        try:
            iId = int(input("Enter student ID: "))
            valid = True
        except:
            print("Error. Invalid ID. Must be numeric.")
    return iId

def FRAmt():
    valid = False
    while valid == False:
        try:
            iFunds = float(input("Enter funds raised: "))
            valid = True
        except:
            print("Error. Invalid amount. Must be a numeric value with decimals.")
    return iFunds

def Major():
    valid = False
    while valid == False:
        try:
            iMaj = int(input("What is your major? (1-Software, 2-Cyber-Security, 3-Interactive Media)"))
            if iMaj == 1:
                iMajor = "Software"
            elif iMaj == 2:
                iMajor = "CyberSecurity"
            elif iMaj ==3:
                iMajor = "Interactive Media"
            if iMaj > 3:
                valid = False
            if iMaj < 1:
                valid = False
            else:
                valid = True
        except:
            print("Error. Enter a 1, 2 or 3.")
    return iMajor

def output(iName, iId, iFunds, iMajor, fundraiser, cTotal):
    print("Funds raised: $", format(cTotal, ',.2f'))
    fundraiser.write(format(iName, '25s') + "\n" + format(iId, '6d') + "\n" + "$" + format(iFunds, '7.2f') + "\n" + format(iMajor, '20s'))

def closing(fundraiser):
    fundraiser.close()

main()
